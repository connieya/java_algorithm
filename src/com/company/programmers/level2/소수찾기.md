# 프로그래머스 Lv.2 소수 찾기

## solution 1

```java
import java.util.*;

class Solution {
    static boolean prime[] = new boolean [10_000_000];
    public int solution(String numbers) {
        Set<Integer> set = new HashSet<>();
        int answer = 0;
        int arr[] = new int[numbers.length()];
        init();
        for(int i=0; i<numbers.length(); i++){
            arr[i] = numbers.charAt(i) - '0';
        }
        Arrays.sort(arr);
        init();
        do {
            String tmp = "";
            for(int i=0; i<arr.length; i++){
                tmp += arr[i];
                int num = Integer.parseInt(tmp);
                if(set.add(num) && !prime[num]){
                    answer++;
                }
            }
            
        }while(next_permutation(arr));
    
        return answer;
    }
    public void init() {
        prime[0] = true;
        prime[1] = true;
        for (int i = 2; i * i < prime.length; i++) {
            if (prime[i]) {
                continue;
            }
            for (int j = i + i; j < prime.length; j += i) {
                prime[j] = true;
            }
        }
    }
    
    public boolean next_permutation(int arr[]) {
        int i = arr.length - 1;
        while(i > 0 && arr[i-1] >= arr[i] ) {
            i--;
        }
        if(i <= 0) return false;
        int j = arr.length -1;
        
        while(arr[i-1] >= arr[j]) {
            j--;
        }
        int temp = arr[i-1];
        arr[i-1] = arr[j];
        arr[j] = temp;
        
        j = arr.length - 1;
        while( i < j) {
            temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
        return true;
    }
    
}
```

## solution 2

```java
import java.util.*;

class Solution {
    Set<Integer> set = new HashSet<>();
    boolean visited[];
    int ans = 0;
    
    boolean isPrime(int number) {
        if(number < 2) return false;
        if(number == 2) return true;
        for(int i=2; i*i <= number; i++) {
            if(number % i == 0) return false;
        }
        return true;
    }
    
    void dfs(int depth , String numbers, String value) {
        if(depth == numbers.length()){
            int a = Integer.parseInt(value);
            if(!set.contains(a) && isPrime(a)){
                set.add(a);
                ans++;
                return;
            } 
         
            return;
        }
        for(int i=0; i<numbers.length(); i++){
            if(!visited[i]){
                visited[i] = true;
                int n = Integer.parseInt(value+numbers.charAt(i));
                if(!set.contains(n) && isPrime(n)){ 
                    set.add(n);
                    ans++;
                }
                dfs(depth+1, numbers, value+numbers.charAt(i));
                visited[i] = false;
            }
        }
    }
    
    public int solution(String numbers) {
        visited = new boolean[numbers.length()];
        dfs(0, numbers , "");
        return ans;
    }
}
```
