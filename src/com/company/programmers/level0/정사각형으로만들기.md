# 프로그래머스 Lv.0 정사각형으로 만들기


## solution 1

```java
class Solution {
    public int[][] solution(int[][] arr) {
        int n = arr.length;
        int m = arr[0].length;
        if(n == m) return arr;
        int[][] answer = {};
        if(n > m)  {
            answer = new int[n][n];
            for(int i=0; i<n; i++) {
                for(int j=0; j<m; j++) {
                    answer[i][j] = arr[i][j];
                }
            }
        }else {
            answer = new int[m][m];
            for(int i=0; i<n; i++){
                for(int j=0; j<m; j++) {
                    answer[i][j] = arr[i][j];
                }
            }
        }

        return answer;
    }
}
```

    public int[][] solution(int[][] arr) {

        int[][] answer = new int[max][max];
        for(int i=0; i<arr.length; i++){
            for(int j=0; j<arr[0].length; j++) {
                answer[i][j] = arr[i][j];
            }
        }
